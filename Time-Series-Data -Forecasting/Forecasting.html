Forecast using nile dataset

data("Nile")
head(Nile)
## [1] 1120 1160  963 1210 1160 1160

convert it into time series data

dat.ts <- ts(Nile,start=1,frequency = 10)
head(dat.ts)
## [1] 1120 1160  963 1210 1160 1160
summary(dat.ts)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##   456.0   798.5   893.5   919.4  1032.0  1370.0

fit a model

fit <- HoltWinters(dat.ts,gamma=F)

Plot data

plot(dat.ts,main="Original data")

plot(fit)

Predict trend

p <- predict(fit, n.ahead=10)
ts.plot(dat.ts,p,main="Predicted with n.ahead as 10")

Summary

  • We have used Holtwinter with Gamma as False
  • The predict says that the trend is going Down

Forecast using AirPassengers dataset

data("AirPassengers")
head(AirPassengers)
## [1] 112 118 132 129 121 135

Plot the data to decide on which model to choose

plot(AirPassengers,main="AirPassagers data")

  • There are cycles
  • Trend in contunially increasing
  • So we use Exponential smoothing
dat1.ts <- ts(AirPassengers,start=1,frequency = 2)
head(dat1.ts)
## [1] 112 118 132 129 121 135

Fit exponential smoothing

fit1 <- HoltWinters(dat1.ts,alpha=0.2, beta=FALSE, gamma=FALSE)
summary(fit1)
##              Length Class  Mode     
## fitted       286    mts    numeric  
## x            144    ts     numeric  
## alpha          1    -none- numeric  
## beta           1    -none- logical  
## gamma          1    -none- logical  
## coefficients   1    -none- numeric  
## seasonal       1    -none- character
## SSE            1    -none- numeric  
## call           5    -none- call

Plot the fitted model

plot(fit1)

Predict trend

p1 <- predict(fit1, n.ahead=10)
ts.plot(dat1.ts,p1,main="Predicted with n.ahead as 10")
